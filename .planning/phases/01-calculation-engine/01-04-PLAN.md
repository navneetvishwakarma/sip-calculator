**packages/core/src/sip.ts** — annuity-due year-by-month loop (beginning-of-period convention):
```typescript
import { monthlyRate } from './utils';
import type { SIPResult, YearlySnapshot } from './schema';

export function calcSIP(
  monthlyAmount: number,
  annualReturnPct: number,
  years: number
): SIPResult {
  if (monthlyAmount <= 0) throw new Error('monthlyAmount must be positive');
  if (annualReturnPct <= 0) throw new Error('annualReturnPct must be positive');
  if (years <= 0 || !Number.isInteger(years) || years > 100) throw new Error('years must be a positive integer between 1 and 100');

  const r = monthlyRate(annualReturnPct);
  let corpus = 0;
  let totalInvested = 0;
  const yearlySnapshots: YearlySnapshot[] = [];

  for (let year = 1; year <= years; year++) {
    for (let month = 1; month <= 12; month++) {
      corpus = (corpus + monthlyAmount) * (1 + r);
      totalInvested += monthlyAmount;
    }
    yearlySnapshots.push({
      year,
      monthlyInvestment: Math.round(monthlyAmount),
      totalInvested: Math.round(totalInvested),
      interestEarned: Math.round(corpus - totalInvested),
      corpusValue: Math.round(corpus),
    });
  }

  return {
    type: 'sip',
    inputs: { monthlyAmount, annualReturnPct, years },
    summary: {
      totalInvested: Math.round(totalInvested),
      totalGains: Math.round(corpus - totalInvested),
      finalCorpus: Math.round(corpus),
    },
    yearlySnapshots,
  };
}
```

CRITICAL — annuity-due convention: `corpus = (corpus + monthlyAmount) * (1 + r)` — contribution is added BEFORE compounding each month (beginning-of-period). This is the standard Indian SIP convention per Groww. Using end-of-period (`corpus = corpus * (1 + r) + monthlyAmount`) will produce ~₹23,000 difference on a ₹23L corpus.

Rounding discipline: intermediate `corpus` float is NEVER rounded between steps. `Math.round()` is only called at `YearlySnapshot` write time and in the final `summary`.

```typescript
import { monthlyRate } from './utils';
import type { StepUpResult, YearlySnapshot } from './schema';

export function calcStepUpSIP(
  monthlyAmount: number,
  annualReturnPct: number,
  years: number,
  stepUpPct: number
): StepUpResult {
  if (monthlyAmount <= 0) throw new Error('monthlyAmount must be positive');
  if (annualReturnPct <= 0) throw new Error('annualReturnPct must be positive');
  if (years <= 0 || !Number.isInteger(years) || years > 100) throw new Error('years must be a positive integer between 1 and 100');
  if (stepUpPct < 0) throw new Error('stepUpPct must be non-negative');

  const r = monthlyRate(annualReturnPct);
  let corpus = 0;
  let totalInvested = 0;
  let currentMonthly = monthlyAmount;
  const yearlySnapshots: YearlySnapshot[] = [];

  for (let year = 1; year <= years; year++) {
    for (let month = 1; month <= 12; month++) {
      corpus = (corpus + currentMonthly) * (1 + r);
      totalInvested += currentMonthly;
    }
    // Capture snapshot BEFORE applying step-up (Risk 2 — Year 1 must show starting amount)
    yearlySnapshots.push({
      year,
      monthlyInvestment: Math.round(currentMonthly),
      totalInvested: Math.round(totalInvested),
      interestEarned: Math.round(corpus - totalInvested),
      corpusValue: Math.round(corpus),
    });
    // Apply step-up AFTER snapshot — takes effect in next year's loop
    currentMonthly = currentMonthly * (1 + stepUpPct / 100);
  }

  return {
    type: 'stepup',
    inputs: { monthlyAmount, stepUpPct, annualReturnPct, years },
    summary: {
      totalInvested: Math.round(totalInvested),
      totalGains: Math.round(corpus - totalInvested),
      finalCorpus: Math.round(corpus),
    },
    yearlySnapshots,
  };
}
```
CRITICAL — step-up timing: `currentMonthly` is incremented AFTER `yearlySnapshots.push()`, not before. This ensures Year 1 snapshot shows the original `monthlyAmount`. See Risk 2 in RESEARCH.md.
