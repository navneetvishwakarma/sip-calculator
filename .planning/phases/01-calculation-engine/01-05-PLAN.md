import { calcSIP } from './sip';
import { calcLumpSum } from './lumpSum';
import type { HybridResult, YearlySnapshot } from './schema';

export function calcHybrid(
  monthlyAmount: number,
  lumpSumAmount: number,
  annualReturnPct: number,
  years: number
): HybridResult {
  if (monthlyAmount <= 0) throw new Error('monthlyAmount must be positive');
  if (lumpSumAmount <= 0) throw new Error('lumpSumAmount must be positive');
  if (annualReturnPct <= 0) throw new Error('annualReturnPct must be positive');
  if (years <= 0 || !Number.isInteger(years)) throw new Error('years must be a positive integer');

  const sipResult = calcSIP(monthlyAmount, annualReturnPct, years);
  const lsResult = calcLumpSum(lumpSumAmount, annualReturnPct, years);

  // Superposition: independent contributions, combine per year
  const yearlySnapshots: YearlySnapshot[] = sipResult.yearlySnapshots.map((sipSnap, i) => {
    const lsSnap = lsResult.yearlySnapshots[i];
    return {
      year: sipSnap.year,
      monthlyInvestment: sipSnap.monthlyInvestment,
      totalInvested: sipSnap.totalInvested + lsSnap.totalInvested,
      interestEarned: sipSnap.interestEarned + lsSnap.interestEarned,
      corpusValue: sipSnap.corpusValue + lsSnap.corpusValue,
    };
  });

  return {
    type: 'hybrid',
    inputs: { monthlyAmount, lumpSumAmount, annualReturnPct, years },
    summary: {
      totalInvested: sipResult.summary.totalInvested + lsResult.summary.totalInvested,
      totalGains: sipResult.summary.totalGains + lsResult.summary.totalGains,
      finalCorpus: sipResult.summary.finalCorpus + lsResult.summary.finalCorpus,
    },
    yearlySnapshots,
  };
}

```typescript
import { monthlyRate } from './utils';
import type { SWPResult, YearlySnapshot } from './schema';

export function calcSWP(
  corpus: number,
  monthlyWithdrawal: number,
  annualReturnPct: number
): SWPResult {
  if (corpus <= 0) throw new Error('corpus must be positive');
  if (monthlyWithdrawal <= 0) throw new Error('monthlyWithdrawal must be positive');
  if (annualReturnPct <= 0) throw new Error('annualReturnPct must be positive');

  const r = monthlyRate(annualReturnPct);
  const inputs = { corpus, monthlyWithdrawal, annualReturnPct };

  // Perpetual check (Pattern 5): monthly growth covers withdrawal forever
  if (corpus * r >= monthlyWithdrawal) {
    return { type: 'swp', inputs, monthsToDepletion: 'perpetual' };
  }

  // Finite depletion loop — track month count and yearly snapshots
  let remaining = corpus;
  let months = 0;
  const yearlySnapshots: YearlySnapshot[] = [];

  while (remaining > 0 && months < 1200) {
    remaining = remaining * (1 + r) - monthlyWithdrawal;
    months++;
    if (months % 12 === 0) {
      const year = months / 12;
      yearlySnapshots.push({
        year,
        monthlyInvestment: 0,
        totalInvested: 0,
        interestEarned: 0,
        corpusValue: Math.max(0, Math.round(remaining)),
      });
    }
  }

  return {
    type: 'swp',
    inputs,
    monthsToDepletion: months,
    yearlySnapshots,
  };
}
```

Note: `months < 1200` safety cap (100-year limit prevents infinite loop for near-perpetual cases below the exact threshold). OQ-1 resolved: `yearlySnapshots` populated for finite depletion cases; omitted (field is optional) for perpetual case.

**packages/core/src/goalSip.ts** — bisection search using closed-form formula for speed, then calcSIP for snapshots:
```typescript
import { monthlyRate } from './utils';
import { calcSIP } from './sip';
import type { GoalResult } from './schema';

export function calcGoalSIP(
  targetCorpus: number,
  annualReturnPct: number,
  years: number
): GoalResult {
  if (targetCorpus <= 0) throw new Error('targetCorpus must be positive');
  if (annualReturnPct <= 0) throw new Error('annualReturnPct must be positive');
  if (years <= 0 || !Number.isInteger(years)) throw new Error('years must be a positive integer');

  const r = monthlyRate(annualReturnPct);
  const n = years * 12;

  // Closed-form annuity-due formula used internally for bisection speed
  // fv(P) = P × ((1+r)^n - 1) / r × (1+r)
  function fv(P: number): number {
    return P * ((Math.pow(1 + r, n) - 1) / r) * (1 + r);
  }

  // Expand upper bound until it exceeds target
  let lo = 0;
  let hi = 1;
  while (fv(hi) < targetCorpus) {
    hi *= 2;
    if (hi > 1e15) throw new Error("targetCorpus is too large — monthly investment would exceed reasonable bounds");
  }

  // Bisection: converge until sub-paisa precision
  while (hi - lo > 0.01) {
    const mid = (lo + hi) / 2;
    if (fv(mid) < targetCorpus) {
      lo = mid;
    } else {
      hi = mid;
    }
  }

  // Round up — ensures monthly amount guarantees meeting the target
  const monthly = Math.ceil((lo + hi) / 2);

  // Produce full yearlySnapshots using the loop-based calcSIP (not the closed-form)
  const sipResult = calcSIP(monthly, annualReturnPct, years);

  return {
    type: 'goal',
    inputs: { targetCorpus, annualReturnPct, years },
    monthlyRequired: monthly,
    summary: sipResult.summary,
    yearlySnapshots: sipResult.yearlySnapshots,
  };
}
```

Note: The bisection uses the closed-form `fv(P)` internally for iteration speed (~22 iterations for ₹1 crore). After convergence, `calcSIP(monthly, ...)` is called once to produce `yearlySnapshots` using the same year-by-month loop as all other calculators. This keeps the computation model consistent across the codebase.

`Math.ceil` ensures the resulting `monthlyRequired` is a whole-rupee amount that meets or exceeds the target. This is ROADMAP success criterion 3 requirement — bisection, not a closed form, is the implementation.
